#NAT
#На HQ-RTR:
object-group network LOCAL_NET
  ip address-range 192.168.100.1-192.168.100.30
exit
nat source
  pool WAN_POOL
    ip address-range 172.16.1.2
  exit
ruleset SNAT
  to interface gi1/0/1 (int to ISP)
  rule 1
    match source-address object-group LOCAL_NET
    action source-nat pool WAN_POOL
    enable
  exit

#На BR-RTR:

  ip address-range 192.168.30.1-192.168.100.14

  pool WAN_POOL
    ip address-range 172.16.2.2
  exit
  ruleset SNAT
  to interface gi1/0/1 (int to ISP)
  rule 1
    match source-address object-group LOCAL_NET
    action source-nat pool WAN_POOL
